<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Behavior Tracker Pro</title>
    <script src="https://cdn.tailwindcss.com"></script>
    <!-- Use the production version of React -->
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <!-- Lucide Icons -->
    <script src="https://unpkg.com/lucide@latest"></script>
</head>
<body>
    <div id="root"></div>
    <script type="text/babel">
       import React, { useState, useMemo, useEffect } from 'react';
import { 
  PlusCircle, MinusCircle, FileText, LayoutDashboard, History, 
  UserPlus, Settings, Trash2, Plus, Pencil, Check, X, 
  Users, Printer, User, Wifi, DownloadCloud, Database, Share2 
} from 'lucide-react';

// Firebase Imports
import { initializeApp } from 'firebase/app';
import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { getFirestore, collection, onSnapshot, doc, setDoc, deleteDoc, query } from 'firebase/firestore';

// Configuration helper
const getFirebaseConfig = () => {
  try {
    return typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};
  } catch (e) {
    return {};
  }
};

export default function App() {
  // Initialize Firebase inside the component once
  const { auth, db, appId } = useMemo(() => {
    const config = getFirebaseConfig();
    const appInstance = initializeApp(config);
    return {
      auth: getAuth(appInstance),
      db: getFirestore(appInstance),
      appId: typeof __app_id !== 'undefined' ? __app_id : 'default-app-id'
    };
  }, []);

  // --- State Management ---
  const [activeTab, setActiveTab] = useState('log');
  const [user, setUser] = useState(null);
  const [isDbConnected, setIsDbConnected] = useState(false);

  // Database Collections
  const [classes, setClasses] = useState([]);
  const [categories, setCategories] = useState([]);
  const [logs, setLogs] = useState([]);
  const [weightingSystem, setWeightingSystem] = useState('multiplier'); 
  const [activeClassId, setActiveClassId] = useState('');

  // UI State - Logging
  const [date, setDate] = useState(new Date().toISOString().split('T')[0]);
  const [selectedStudent, setSelectedStudent] = useState('');
  const [selectedCategoryId, setSelectedCategoryId] = useState('');
  const [selectedSubcategory, setSelectedSubcategory] = useState('');
  const [notes, setNotes] = useState('');

  // UI State - Classes & Students
  const [newClassName, setNewClassName] = useState('');
  const [studentInputs, setStudentInputs] = useState({}); 
  const [reportStudent, setReportStudent] = useState('');

  // UI State - Categories
  const [newCatName, setNewCatName] = useState('');
  const [newCatWeight, setNewCatWeight] = useState(1);
  const [subCatInputs, setSubCatInputs] = useState({});

  // --- 1. Authentication (Rule 3) ---
  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (err) {
        console.error("Auth Error:", err);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (u) => setUser(u));
    return () => unsubscribe();
  }, [auth]);

  // --- 2. Real-time Database Syncing (Rules 1 & 2) ---
  useEffect(() => {
    if (!user) return;

    const classesRef = collection(db, 'artifacts', appId, 'public', 'data', 'classes');
    const catsRef = collection(db, 'artifacts', appId, 'public', 'data', 'categories');
    const logsRef = collection(db, 'artifacts', appId, 'public', 'data', 'logs');
    const settingsRef = collection(db, 'artifacts', appId, 'public', 'data', 'settings');

    const unsubClasses = onSnapshot(classesRef, (snap) => {
      const dbClasses = snap.docs.map(d => d.data());
      setClasses(dbClasses);
      if (dbClasses.length > 0 && !activeClassId) setActiveClassId(dbClasses[0].id);
      setIsDbConnected(true);
    }, console.error);

    const unsubCats = onSnapshot(catsRef, (snap) => {
      const dbCats = snap.docs.map(d => d.data());
      setCategories(dbCats);
      if (dbCats.length > 0 && !selectedCategoryId) setSelectedCategoryId(dbCats[0].id);
    }, console.error);

    const unsubLogs = onSnapshot(logsRef, (snap) => {
      const sorted = snap.docs.map(d => d.data()).sort((a,b) => b.id - a.id);
      setLogs(sorted);
    }, console.error);

    const unsubSettings = onSnapshot(settingsRef, (snap) => {
      const gSet = snap.docs.find(d => d.id === 'global')?.data();
      if (gSet?.weightingSystem) setWeightingSystem(gSet.weightingSystem);
    }, console.error);

    return () => { 
      unsubClasses(); 
      unsubCats(); 
      unsubLogs(); 
      unsubSettings(); 
    };
  }, [user, appId, db, activeClassId, selectedCategoryId]);

  // Derived Data
  const currentClass = classes.find(c => c.id === activeClassId);
  const currentStudents = currentClass ? (currentClass.students || []) : [];
  const currentCategory = categories.find(c => c.id === selectedCategoryId);
  const currentSubcategories = currentCategory ? (currentCategory.subcategories || []) : [];

  // Data Selectors Management
  useEffect(() => {
    if (currentStudents.length > 0) {
      if (!currentStudents.includes(selectedStudent)) setSelectedStudent(currentStudents[0]);
      if (!currentStudents.includes(reportStudent)) setReportStudent(currentStudents[0]);
    }
  }, [activeClassId, currentStudents]);

  useEffect(() => {
    if (currentSubcategories.length > 0) {
      if (!currentSubcategories.includes(selectedSubcategory)) setSelectedSubcategory(currentSubcategories[0]);
    } else {
      setSelectedSubcategory('');
    }
  }, [selectedCategoryId, currentSubcategories]);

  // Point Calculation Logic
  const calculateFinalPoints = (raw, weight, system) => {
    return system === 'percentage' 
      ? Number((raw * (weight / 100)).toFixed(2)) 
      : Number((raw * weight).toFixed(2));
  };

  // --- Handlers ---
  const handleShare = () => {
    const shareUrl = `${window.location.origin}/artifacts/${appId}/public/index.html`;
    const subject = encodeURIComponent("Collaborate on Behavior Tracker Pro");
    const body = encodeURIComponent(`Access our live behavior tracking dashboard here:\n\n${shareUrl}`);
    window.location.href = `mailto:?subject=${subject}&body=${body}`;
  };

  const processNewLog = async (pts) => {
    if (!user || !selectedStudent || !currentCategory) return;

    const finalPts = calculateFinalPoints(pts, currentCategory.weight, weightingSystem);
    const newId = Date.now().toString();

    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'logs', newId), {
      id: newId,
      date,
      classId: activeClassId,
      student: selectedStudent,
      categoryId: currentCategory.id,
      categoryName: currentCategory.name,
      subcategory: selectedSubcategory || '',
      rawPoints: pts,
      weight: currentCategory.weight,
      finalPoints: finalPts,
      notes
    });
    setNotes(''); 
  };

  const handleAddClass = async (e) => {
    e.preventDefault();
    if (!user || !newClassName.trim()) return;
    const newId = `c${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'classes', newId), {
      id: newId, name: newClassName.trim(), students: []
    });
    setNewClassName('');
  };

  const handleAddStudent = async (classId) => {
    if (!user) return;
    const name = studentInputs[classId]?.trim();
    if (!name || !currentClass) return;
    const updatedStudents = [...(currentClass.students || [])];
    if (!updatedStudents.includes(name)) {
      updatedStudents.push(name);
      await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'classes', classId), {
        ...currentClass, students: updatedStudents.sort()
      });
    }
    setStudentInputs({ ...studentInputs, [classId]: '' });
  };

  const handleAddCategory = async (e) => {
    e.preventDefault();
    if (!user || !newCatName.trim()) return;
    const id = `cat${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'categories', id), {
      id, name: newCatName.trim(), weight: parseFloat(newCatWeight) || 1, subcategories: []
    });
    setNewCatName('');
  };

  const handleAddSubcategory = async (catId) => {
    if (!user) return;
    const subName = subCatInputs[catId]?.trim();
    if (!subName) return;
    const cat = categories.find(c => c.id === catId);
    if (cat) {
      const subs = [...(cat.subcategories || [])];
      if (!subs.includes(subName)) {
        subs.push(subName);
        await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'categories', catId), {
          ...cat, subcategories: subs.sort()
        });
      }
    }
    setSubCatInputs({ ...subCatInputs, [catId]: '' });
  };

  const updateWeightingSystem = async (system) => {
    if (!user) return;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'settings', 'global'), { weightingSystem: system });
  };

  const exportToCSV = () => {
    const headers = ['Date', 'Student', 'Category', 'Subcategory', 'Raw Points', 'Net Points', 'Notes'];
    const filteredLogs = logs.filter(l => l.classId === activeClassId);
    const rows = filteredLogs.map(l => [
      l.date, l.student, l.categoryName, l.subcategory, l.rawPoints, l.finalPoints, `"${(l.notes || '').replace(/"/g, '""')}"`
    ]);
    const csvContent = "data:text/csv;charset=utf-8," + [headers.join(','), ...rows.map(r => r.join(','))].join('\n');
    const encodedUri = encodeURI(csvContent);
    const link = document.createElement("a");
    link.setAttribute("href", encodedUri);
    link.setAttribute("download", `behavior_logs_${activeClassId}.csv`);
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  // --- Computed Views ---
  const filteredLogs = useMemo(() => logs.filter(l => l.classId === activeClassId), [logs, activeClassId]);
  
  const dashboardData = useMemo(() => {
    const data = {};
    currentStudents.forEach(s => data[s] = { total: 0 });
    filteredLogs.forEach(l => {
      if (data[l.student]) data[l.student].total += l.finalPoints;
    });
    return Object.entries(data).map(([student, s]) => ({ student, total: Number(s.total.toFixed(2)) }));
  }, [filteredLogs, currentStudents]);

  // Derived Data for Reports
  const studentLogs = useMemo(() => filteredLogs.filter(l => l.student === reportStudent), [filteredLogs, reportStudent]);
  const studentTotal = studentLogs.reduce((sum, l) => sum + l.finalPoints, 0);
  const categoryTotals = useMemo(() => {
    const totals = {};
    studentLogs.forEach(l => {
      totals[l.categoryName] = (totals[l.categoryName] || 0) + l.finalPoints;
    });
    return Object.entries(totals).map(([name, total]) => ({ name, total: Number(total.toFixed(2)) }));
  }, [studentLogs]);

  if (!user) return <div className="min-h-screen flex items-center justify-center bg-gray-50"><p className="text-gray-500 animate-pulse">Initializing Live Session...</p></div>;

  return (
    <div className="min-h-screen bg-gray-100 text-gray-800 font-sans p-4 md:p-8">
      <div className="max-w-6xl mx-auto">
        <header className="mb-6 flex flex-col md:flex-row md:items-center justify-between gap-4">
          <div>
            <h1 className="text-3xl font-bold text-gray-900 flex items-center gap-2">
              <LayoutDashboard className="text-blue-600" /> Behavior Tracker Pro
            </h1>
            <div className="flex items-center gap-3 mt-1">
              <span className={`text-xs font-bold px-2 py-0.5 rounded-full border flex items-center gap-1 ${isDbConnected ? 'text-green-700 bg-green-100 border-green-200' : 'text-amber-700 bg-amber-100 border-amber-200'}`}>
                <Wifi size={12}/> {isDbConnected ? 'Live Sync Active' : 'Connecting...'}
              </span>
            </div>
          </div>
          <div className="flex items-center gap-3">
            <button onClick={handleShare} className="bg-white border p-2 rounded-lg text-blue-600 hover:bg-blue-50 transition" title="Share via Email"><Share2 size={20}/></button>
            <div className="bg-white p-2 rounded-lg shadow-sm border flex items-center gap-2">
              <Users size={18} className="text-gray-400" />
              <select value={activeClassId} onChange={(e) => setActiveClassId(e.target.value)} className="bg-transparent border-none font-semibold focus:ring-0 outline-none">
                {classes.map(c => <option key={c.id} value={c.id}>{c.name}</option>)}
                {classes.length === 0 && <option value="">No Classes</option>}
              </select>
            </div>
          </div>
        </header>

        <nav className="flex gap-2 mb-6 bg-white p-1 rounded-lg shadow-sm w-fit overflow-x-auto no-scrollbar">
          {['log', 'dashboard', 'reports', 'classes', 'settings'].map(tab => (
            <button key={tab} onClick={() => setActiveTab(tab)} className={`px-4 py-2 rounded-md capitalize transition whitespace-nowrap ${activeTab === tab ? 'bg-blue-600 text-white shadow-md' : 'text-gray-600 hover:bg-gray-100'}`}>
              {tab === 'log' ? 'Daily Log' : tab}
            </button>
          ))}
        </nav>

        {activeTab === 'log' && (
          <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
            <div className="bg-white p-6 rounded-xl shadow-sm border h-fit sticky top-4">
              <h2 className="text-lg font-bold mb-4 flex items-center gap-2 text-blue-600"><PlusCircle size={20}/> New Entry</h2>
              <div className="space-y-4">
                <div>
                  <label className="text-xs font-bold text-gray-400 uppercase">Student</label>
                  <select className="w-full border p-2 rounded bg-gray-50 mt-1" value={selectedStudent} onChange={e => setSelectedStudent(e.target.value)}>
                    {currentStudents.map(s => <option key={s} value={s}>{s}</option>)}
                    {!currentStudents.length && <option>No students in class</option>}
                  </select>
                </div>
                
                <div>
                  <label className="text-xs font-bold text-gray-400 uppercase">Category</label>
                  <select className="w-full border p-2 rounded bg-gray-50 mt-1" value={selectedCategoryId} onChange={e => setSelectedCategoryId(e.target.value)}>
                    {categories.map(c => <option key={c.id} value={c.id}>{c.name}</option>)}
                    {!categories.length && <option>No categories setup</option>}
                  </select>
                </div>

                {currentSubcategories.length > 0 && (
                  <div>
                    <label className="text-xs font-bold text-gray-400 uppercase">Subcategory</label>
                    <select className="w-full border p-2 rounded bg-gray-50 mt-1" value={selectedSubcategory} onChange={e => setSelectedSubcategory(e.target.value)}>
                      {currentSubcategories.map(s => <option key={s} value={s}>{s}</option>)}
                    </select>
                  </div>
                )}

                <textarea className="w-full border p-2 rounded bg-gray-50 h-24 mt-2" placeholder="Notes..." value={notes} onChange={e => setNotes(e.target.value)} />
                
                <div className="flex gap-2">
                  <button onClick={() => processNewLog(1)} className="flex-1 bg-green-500 text-white py-3 rounded-lg font-bold hover:bg-green-600 transition shadow-sm">+1 Point</button>
                  <button onClick={() => processNewLog(-1)} className="flex-1 bg-red-500 text-white py-3 rounded-lg font-bold hover:bg-red-600 transition shadow-sm">-1 Point</button>
                </div>
              </div>
            </div>

            <div className="lg:col-span-2 space-y-4">
              <div className="bg-white rounded-xl shadow-sm border overflow-hidden">
                <div className="p-4 bg-gray-50 border-b flex justify-between items-center">
                  <h3 className="font-bold flex items-center gap-2 text-gray-700"><History size={18}/> Live Feed</h3>
                  <button onClick={exportToCSV} className="text-blue-600 text-sm font-bold flex items-center gap-1 hover:underline">
                    <DownloadCloud size={16}/> Export CSV
                  </button>
                </div>
                <div className="overflow-x-auto">
                  <table className="w-full text-left">
                    <thead className="bg-white border-b">
                      <tr>
                        <th className="p-3 text-xs uppercase text-gray-400">Student</th>
                        <th className="p-3 text-xs uppercase text-gray-400">Pts</th>
                        <th className="p-3 text-xs uppercase text-gray-400">Category</th>
                        <th className="p-3 text-xs uppercase text-gray-400">Actions</th>
                      </tr>
                    </thead>
                    <tbody className="divide-y">
                      {filteredLogs.map(l => (
                        <tr key={l.id} className="hover:bg-gray-50 group">
                          <td className="p-3">
                            <div className="font-medium text-gray-900">{l.student}</div>
                            <div className="text-[10px] text-gray-400">{new Date(parseInt(l.id)).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}</div>
                          </td>
                          <td className="p-3">
                            <span className={`font-bold px-2 py-0.5 rounded text-xs ${l.finalPoints >= 0 ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'}`}>
                              {l.finalPoints > 0 ? '+' : ''}{l.finalPoints}
                            </span>
                          </td>
                          <td className="p-3">
                            <div className="text-sm font-medium">{l.categoryName}</div>
                            {l.subcategory && <div className="text-xs text-blue-500 italic">{l.subcategory}</div>}
                          </td>
                          <td className="p-3">
                            <button onClick={() => deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'logs', l.id))} className="text-gray-300 hover:text-red-500 transition opacity-0 group-hover:opacity-100">
                              <Trash2 size={16}/>
                            </button>
                          </td>
                        </tr>
                      ))}
                      {filteredLogs.length === 0 && <tr><td colSpan="4" className="p-8 text-center text-gray-400">No logs for this class yet.</td></tr>}
                    </tbody>
                  </table>
                </div>
              </div>
            </div>
          </div>
        )}

        {activeTab === 'reports' && (
          <div className="space-y-6">
            <div className="bg-white p-6 rounded-xl shadow-sm border print:shadow-none print:border-none print:p-0">
              <div className="flex justify-between items-center mb-6 print:hidden">
                <h2 className="text-xl font-bold flex items-center gap-2"><User size={22}/> Student Report</h2>
                <button onClick={() => window.print()} className="bg-blue-50 text-blue-600 px-4 py-2 rounded-lg font-bold flex items-center gap-2 hover:bg-blue-100 transition">
                  <Printer size={18}/> Print Report
                </button>
              </div>

              <div className="mb-6 print:mb-4">
                <label className="text-xs font-bold text-gray-400 uppercase print:hidden">Select Student</label>
                <select className="w-full md:w-1/3 border p-2 rounded bg-gray-50 mt-1 print:hidden" value={reportStudent} onChange={e => setReportStudent(e.target.value)}>
                  {currentStudents.map(s => <option key={s} value={s}>{s}</option>)}
                  {!currentStudents.length && <option>No students</option>}
                </select>
                <h3 className="hidden print:block text-2xl font-black mb-1">Behavior Report: {reportStudent}</h3>
                <p className="hidden print:block text-sm text-gray-500 mb-4">Class: {currentClass?.name} | Date Generated: {new Date().toLocaleDateString()}</p>
              </div>

              {reportStudent ? (
                <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
                  <div className="md:col-span-1 space-y-6">
                    <div className="bg-gray-50 p-6 rounded-xl border text-center print:border-gray-200">
                      <div className="text-sm font-bold text-gray-500 uppercase mb-2">Net Points</div>
                      <div className={`text-5xl font-black ${studentTotal >= 0 ? 'text-green-600' : 'text-red-600'}`}>
                        {studentTotal > 0 ? '+' : ''}{Number(studentTotal.toFixed(2))}
                      </div>
                    </div>

                    <div className="bg-gray-50 p-4 rounded-xl border print:border-gray-200">
                      <h4 className="font-bold text-gray-700 mb-3 text-sm uppercase">By Category</h4>
                      <div className="space-y-2">
                        {categoryTotals.map(c => (
                          <div key={c.name} className="flex justify-between items-center text-sm">
                            <span className="text-gray-600">{c.name}</span>
                            <span className={`font-bold ${c.total >= 0 ? 'text-green-600' : 'text-red-600'}`}>{c.total > 0 ? '+' : ''}{c.total}</span>
                          </div>
                        ))}
                        {categoryTotals.length === 0 && <p className="text-xs text-gray-400 italic">No category data.</p>}
                      </div>
                    </div>
                  </div>

                  <div className="md:col-span-2">
                    <h4 className="font-bold text-gray-700 mb-3 text-sm uppercase">Log History</h4>
                    <div className="overflow-x-auto border rounded-xl print:border-gray-200">
                      <table className="w-full text-left text-sm">
                        <thead className="bg-gray-50 border-b">
                          <tr>
                            <th className="p-3 text-xs uppercase text-gray-500">Date/Time</th>
                            <th className="p-3 text-xs uppercase text-gray-500">Pts</th>
                            <th className="p-3 text-xs uppercase text-gray-500">Category</th>
                            <th className="p-3 text-xs uppercase text-gray-500">Notes</th>
                          </tr>
                        </thead>
                        <tbody className="divide-y">
                          {studentLogs.map(l => (
                            <tr key={l.id} className="bg-white">
                              <td className="p-3">
                                <div className="font-medium">{l.date}</div>
                                <div className="text-[10px] text-gray-400">{new Date(parseInt(l.id)).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}</div>
                              </td>
                              <td className="p-3">
                                <span className={`font-bold px-2 py-0.5 rounded text-xs ${l.finalPoints >= 0 ? 'bg-green-100 text-green-700 print:text-black print:bg-transparent print:p-0' : 'bg-red-100 text-red-700 print:text-black print:bg-transparent print:p-0'}`}>
                                  {l.finalPoints > 0 ? '+' : ''}{l.finalPoints}
                                </span>
                              </td>
                              <td className="p-3">
                                <div className="font-medium">{l.categoryName}</div>
                                {l.subcategory && <div className="text-[10px] text-blue-500 italic print:text-gray-600">{l.subcategory}</div>}
                              </td>
                              <td className="p-3 text-gray-600 text-xs">{l.notes || '-'}</td>
                            </tr>
                          ))}
                          {studentLogs.length === 0 && <tr><td colSpan="4" className="p-6 text-center text-gray-400">No logs found for {reportStudent}.</td></tr>}
                        </tbody>
                      </table>
                    </div>
                  </div>
                </div>
              ) : (
                <div className="text-center p-12 text-gray-400 border-2 border-dashed rounded-xl print:hidden">
                  Select a student to view their report.
                </div>
              )}
            </div>
          </div>
        )}

        {activeTab === 'classes' && (
          <div className="space-y-6">
            <div className="bg-white p-6 rounded-xl border shadow-sm">
              <h3 className="font-bold mb-4 flex items-center gap-2"><Plus size={18}/> Create Class</h3>
              <form onSubmit={handleAddClass} className="flex gap-2">
                <input className="flex-1 border p-2 rounded-lg" placeholder="e.g., 3rd Period Math" value={newClassName} onChange={e => setNewClassName(e.target.value)} />
                <button className="bg-blue-600 text-white px-6 rounded-lg font-bold hover:bg-blue-700 transition">Create</button>
              </form>
            </div>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
              {classes.map(c => (
                <div key={c.id} className="bg-white p-6 rounded-xl border shadow-sm flex flex-col">
                  <div className="flex justify-between items-center mb-4">
                    <h4 className="text-lg font-bold text-gray-800">{c.name}</h4>
                    <button onClick={() => deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'classes', c.id))} className="text-gray-400 hover:text-red-500 transition"><Trash2 size={18}/></button>
                  </div>
                  <div className="flex gap-2 mb-4">
                    <input className="flex-1 border p-2 rounded-lg text-sm" placeholder="Add student..." value={studentInputs[c.id] || ''} onChange={e => setStudentInputs({...studentInputs, [c.id]: e.target.value})} onKeyDown={e => e.key === 'Enter' && handleAddStudent(c.id)} />
                    <button onClick={() => handleAddStudent(c.id)} className="bg-gray-800 text-white px-4 rounded-lg text-sm font-bold">Add</button>
                  </div>
                  <div className="flex flex-wrap gap-2 flex-1">
                    {(c.students || []).map(s => (
                      <span key={s} className="bg-gray-100 text-gray-600 px-2 py-1 rounded-md text-xs font-medium flex items-center gap-2 group">
                        {s}
                        <button onClick={() => {
                          const updated = c.students.filter(x => x !== s);
                          setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'classes', c.id), {...c, students: updated});
                        }} className="text-gray-400 hover:text-red-500"><X size={12}/></button>
                      </span>
                    ))}
                    {(!c.students || c.students.length === 0) && <p className="text-xs text-gray-400 italic py-2">No students added yet.</p>}
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        {activeTab === 'settings' && (
          <div className="space-y-6">
            <div className="bg-white p-6 rounded-xl border shadow-sm">
              <h3 className="font-bold mb-4 flex items-center gap-2"><Settings size={18}/> Weighting System</h3>
              <div className="flex gap-4 p-4 bg-gray-50 rounded-xl">
                <button 
                  onClick={() => updateWeightingSystem('multiplier')}
                  className={`flex-1 p-3 rounded-lg border-2 transition text-center ${weightingSystem === 'multiplier' ? 'border-blue-600 bg-blue-50 text-blue-700' : 'border-transparent bg-white text-gray-500'}`}
                >
                  <div className="font-bold">Multiplier</div>
                  <div className="text-[10px]">Points x Weight (e.g., 1pt x 2.0)</div>
                </button>
                <button 
                  onClick={() => updateWeightingSystem('percentage')}
                  className={`flex-1 p-3 rounded-lg border-2 transition text-center ${weightingSystem === 'percentage' ? 'border-blue-600 bg-blue-50 text-blue-700' : 'border-transparent bg-white text-gray-500'}`}
                >
                  <div className="font-bold">Percentage</div>
                  <div className="text-[10px]">Points x (Weight/100)</div>
                </button>
              </div>
            </div>

            <div className="bg-white p-6 rounded-xl border shadow-sm">
              <h3 className="font-bold mb-6 flex items-center gap-2 text-gray-700"><FileText size={20}/> Categories & Subcategories</h3>
              <form onSubmit={handleAddCategory} className="flex flex-col md:flex-row gap-3 mb-8 bg-blue-50 p-4 rounded-xl border border-blue-100">
                <div className="flex-1">
                  <label className="text-[10px] font-bold text-blue-400 uppercase ml-1">Category Name</label>
                  <input className="w-full border p-2 rounded-lg mt-1" placeholder="e.g., Participation" value={newCatName} onChange={e => setNewCatName(e.target.value)} />
                </div>
                <div className="w-32">
                  <label className="text-[10px] font-bold text-blue-400 uppercase ml-1">Weight</label>
                  <input className="w-full border p-2 rounded-lg mt-1" type="number" step="0.5" value={newCatWeight} onChange={e => setNewCatWeight(e.target.value)} />
                </div>
                <button className="bg-blue-600 text-white px-6 rounded-lg font-bold hover:bg-blue-700 transition self-end h-10">Add</button>
              </form>

              <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                {categories.map(cat => (
                  <div key={cat.id} className="p-4 border rounded-xl bg-gray-50 flex flex-col">
                    <div className="flex justify-between items-start mb-4">
                      <div>
                        <span className="font-bold text-gray-800">{cat.name}</span>
                        <div className="text-xs font-bold text-blue-600 bg-blue-100 px-2 rounded-full w-fit mt-1">Weight: {cat.weight}</div>
                      </div>
                      <button onClick={() => deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'categories', cat.id))} className="text-gray-300 hover:text-red-500 transition"><Trash2 size={16}/></button>
                    </div>
                    
                    <div className="flex gap-2 mb-3">
                      <input className="flex-1 border p-1.5 rounded-lg text-xs" placeholder="Add subcategory..." value={subCatInputs[cat.id] || ''} onChange={e => setSubCatInputs({...subCatInputs, [cat.id]: e.target.value})} />
                      <button onClick={() => handleAddSubcategory(cat.id)} className="bg-gray-800 text-white px-3 rounded-lg text-[10px] font-bold uppercase">Add</button>
                    </div>

                    <div className="flex flex-wrap gap-1.5">
                      {(cat.subcategories || []).map(sub => (
                        <span key={sub} className="bg-white border text-gray-500 px-2 py-0.5 rounded-full text-[10px] flex items-center gap-1 group">
                          {sub}
                          <button onClick={() => {
                            const updated = cat.subcategories.filter(x => x !== sub);
                            setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'categories', cat.id), {...cat, subcategories: updated});
                          }} className="hover:text-red-500 text-gray-300"><X size={10}/></button>
                        </span>
                      ))}
                    </div>
                  </div>
                ))}
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
